NgRx Actions

Definition

Actions are one of the main building blocks in NgRx. Actions express unique events that happen throughout your application.

Actions are objects that have a type property and optionally a payload. They describe what happened but don't specify how the application state changes.

Actions are dispatched from components and effects to trigger state changes in reducers.

Action Implementations
Action Definition
// actions.ts
import {{ '{' }} createAction, props } from '@ngrx/store';

export const addItem = createAction(
  '[Item] Add Item',
  props<{{ '{' }} item: string }>()
);

export const removeItem = createAction(
  '[Item] Remove Item',
  props<{{ '{' }} index: number }>()
);

export const clearItems = createAction(
  '[Item] Clear All Items'
);
Dispatching Actions
// component.ts
import {{ '{' }} Store } from '@ngrx/store';
import * as ItemActions from './actions';

constructor(private store: Store) {{ '{' }}}

addNewItem() {{ '{' }}
  this.store.dispatch(
    ItemActions.addItem({{ '{' }} 
      item: 'New Item' 
    }})
  );
}

removeItemAt(index: number) {{ '{' }}
  this.store.dispatch(
    ItemActions.removeItem({{ '{' }} index }})
  );
}
